switch判斷式

switch判斷式是用來做多重條件判斷的

switch (表達式) {
    case 值1:
        // 當表達式的值等於值1時執行的程式
        break;
    case 值2:
        // 當表達式的值等於值2時執行的程式
        break;
    ...
    default:
        // 當表達式的值不符合任何case時執行的程式
        break;
}

迴圈

迴圈是用來重複執行程式碼的結構

for (初始化計數器; 執行條件; 步進值) {
    // 重複執行的程式碼
}

for/in

for/in 針對陣列的項目作處理

for (變數 in 陣列) {
    // 針對陣列的每個項目執行的程式碼
}

while 迴圈

while (執行的條件) {
    // 重複執行的程式碼
}

do...while 迴圈

do...while 迴圈至少會執行一次，然後再檢查條件是否繼續執行

do {
    // 重複執行的程式碼
} while (執行的條件);

break與continue

break;    // 用來跳出迴圈或switch判斷式
          for (let i = 0; i < 10; i++) {
              if (i === 5) {
                  break; // 當i等於5時跳出迴圈
              }
          }
          // 其他程式碼

continue; // 用來跳過本次迴圈，進入下一次迴圈
            for (let i = 0; i < 10; i++) {
                if (i % 2 === 0) {
                    continue; // 當i是偶數時跳過本次迴圈
                }
                // 其他程式碼
            }
